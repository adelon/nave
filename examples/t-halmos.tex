% Original:
% The \textit{tensor product} $U\otimes V$
% of two finite-dimensional spaces $U$ and $V$ (over the same field)
% is the dual of the vector space of all bilinear forms on $U\oplus V$.

\begin{definition}
   Let $K$ be a field and let $U, V$ be vector spaces over $K$.
   The tensor product of $U$ and $V$, $U\tensor V$,
   is the dual of the space of bilinear forms on $U\directSum V$.
\end{definition}
